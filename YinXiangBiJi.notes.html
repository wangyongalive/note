var diameterOfBinaryTree = function (root) {

    let ans = 0;

    function depth(rootNode) {

        if (!rootNode) {

            // 如果不存在根节点，则深度为0

            return 0;

        }

        // 递归，获取左子树的深度

        let L = depth(rootNode.left);

        // 递归，获取右子树的深度

        let R = depth(rootNode.right);

        ans = Math.max(ans, L + R);

        return Math.max(L, R) + 1;

    }

    depth(root);

    return ans;

};

class MinHeap {

    constructor() {

        this.heap = [];

    }

    swap(i1, i2) {

        [this.heap[i1], this.heap[i2]] = [this.heap[i2], this.heap[i1]]

    }

    getParentIndex(i) {

        return (i - 1) >> 1; // 相对于除以2 取整

    }

    getLeftIndex(i) {

        return i * 2 + 1

    }

    getRightIndex(i) {

        return i * 2 + 2

    }



    shiftUp(index) {

        if (index == 0) return; // 到堆顶了

        const parentIndex = this.getParentIndex(index);

        if (this.heap[parentIndex] > this.heap[index]) {

            this.swap(parentIndex, index)

            this.shiftUp(parentIndex);

        }

    }

    shiftDown(index) {

        const leftIndex = this.getLeftIndex(index);

        const rightIndex = this.getRightIndex(index);

        if (this.heap[leftIndex] < this.heap[index]) {

            this.swap(leftIndex, index);

            this.shiftDown(leftIndex);

        }

        if (this.heap[rightIndex] < this.heap[index]) {

            this.swap(rightIndex, index);

            this.shiftDown(rightIndex);

        }

    }

    // 插入

    add(value) {

        this.heap.push(value);

        this.shiftUp(this.heap.length - 1);

    }

    pop() {

        this.heap[0] = this.heap.pop();

        this.shiftDown(0);

    }

    peek() {

        return this.heap[0];

    }

    size() {

        return this.heap.length;

    }

}



var KthLargest = function (k, nums) {

    this.k = k;

    this.heap = new MinHeap(); // 最小堆

    for (const x of nums) {

        this.add(x);

    }

};



KthLargest.prototype.add = function (val) {

    if (this.heap.size() > this.k) {

        this.heap.pop();

    }

    return this.heap.peek();

var longestPalindrome = function (s) {

    let len = s.length;

    if (len < 2) {

        return s;

    }

    let maxLen = 1, start = 0;

    let dp = Array.from(new Array(len), (item) => new Array(len).fill(0)); // 二维数组

    for (let i = 0; i < len; i++) {

        dp[i][i] = true;

    }

    //  按照列顺序填充dp

    for (let j = 1; j < len; j++) {

        for (let i = 0; i < j; i++) {

            if (s[i] !== s[j]) {

                dp[i][j] = false;

            } else {

                if (j - i <= 2) {

                    dp[i][j] = true; // 3或者2个字符串是回文

                } else {

                    dp[i][j] = dp[i + 1][j - 1];

                }

            }

            // console.log(dp)

            if (j - i + 1 > maxLen && dp[i][j]) {

                maxLen = j - i + 1;

                start = i;

            }

        }

    }

    // console.log(start,maxLen)

    return s.substr(start, maxLen)

};
