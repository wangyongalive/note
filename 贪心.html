 * @param {number[]} s

 * @return {number}

 */

var findContentChildren = function (g, s) {

    const sortFunc = (a, b) => a - b

    g.sort(sortFunc);

    s.sort(sortFunc);

    let i = 0;

    s.forEach(n => {

        if (n >= g[i]) {

            i++;

        }

    })

    return i;

};

var wiggleMaxLength = function (nums) {

    if (nums.length < 2) return nums.length

    let trend, len = 1, i = 0

    while (++i < nums.length)

        if (nums[i] !== nums[i - 1])

            if (nums[i] > nums[i - 1] !== trend) { // 本轮趋势

                len++

                trend = nums[i] > nums[i - 1] // 更改趋势

            }

    return len

};

var canJump = function (nums) {

    const len = nums.length;

    let rightmost = 0;

    for (let i = 0; i < len; i++) {

        if (i <= rightmost) {

            rightmost = Math.max(rightmost, i + nums[i]); // 取覆盖区间的最大值

            if (rightmost >= len - 1) {

                return true;

            }

        }

    }

    return false;

};

/**

 * @return {number}

 */

var jump = function (nums) {

    let curIndex = 0,  

        nextIndex = 0, // 最远距离

        steps = 0; // 步数

    for (let i = 0; i < nums.length - 1; i++) { // 注意这里是小于 nums.length - 1

        nextIndex = Math.max(nums[i] + i, nextIndex)

        if (i === curIndex) {

            curIndex = nextIndex

            steps++

        }

    }

    return steps

};

/**

 * @param {number[]} nums

 * @param {number} k

 * @return {number}

 */

var largestSumAfterKNegations = function (nums, k) {

    // 将数组按照绝对值大小从大到小排序，注意要按照绝对值的大小

    nums.sort((a, b) => Math.abs(b) - Math.abs(a))

    // 从前向后遍历，遇到负数将其变为正数，同时K--

    for (let i = 0; i < nums.length; i++) {

        if (nums[i] < 0 && k > 0) {

            nums[i] *= -1

            k--

        }

    }

   // 如果K还大于0，那么反复转变数值最小的元素，将K用完

    if (k > 0 && k % 2 === 1) {

        nums[nums.length - 1] *= -1

    }

    k = 0

    // 

    return nums.reduce((a, b) => a + b)

};

/**

 * @param {number[]} gas

 * @param {number[]} cost

 * @return {number}

 */

var canCompleteCircuit = function (gas, cost) {

    const gasLen = gas.length

    let start = 0

    let curSum = 0

    let totalSum = 0

    for (let i = 0; i < gasLen; i++) {

        curSum += gas[i] - cost[i]

        totalSum += gas[i] - cost[i]

        if (curSum < 0) {// 当前累加rest[i]和 curSum一旦小于0

            curSum = 0  // curSum从0开始

            start = i + 1 // 起始位置更新为i+1

        }

    }

    if (totalSum < 0) return -1 // 说明怎么走都不可能跑一圈了

    return start

};

var lemonadeChange = function (bills) {

    let fiveCount = 0, tenCount = 0

    for (bill of bills) {

        if (bill === 5) {

            fiveCount += 1

        } else if (bill === 10) {

            if (fiveCount > 0) {

                fiveCount -= 1

                tenCount += 1

            } else {

                return false

            }

        } else {

            if (tenCount > 0 && fiveCount > 0) { // 贪心

                tenCount -= 1

                fiveCount -= 1

            } else if (fiveCount >= 3) {

                fiveCount -= 3

            } else {

                return false

            }

        }

    }

    return true

};

var reconstructQueue = function (people) {

    let queue = []

    people.sort((a, b) => {

        if (b[0] !== a[0]) {

            return b[0] - a[0]

        } else {

            return a[1] - b[1]

        }

    })

    // console.log(people)

    // 插入操作

    for (let i = 0; i < people.length; i++) {

        queue.splice(people[i][1], 0, people[i])

    }

    return queue

};

/**

 * @param {number[][]} points

 * @return {number}

 */

 // 右边界

function findMinArrowShots(points) {

    if (!points.length) return 0;

    points.sort((a, b) => a[1] - b[1]); // 按照结束坐标排序

    let count = 1; // 最少有一个区间不相交

    let xEnd = points[0][1];

    for (let item of points) {

        const start = item[0];

        // 这里是> 或者 >=取决于, [1,3][3,6], 3算不算相交

        if (start > xEnd) {

            count++;

            xEnd = item[1];

        }

    }

    return count;

}

/**

 */

var partitionLabels = function (s) {

    let hash = {}

    for (let i = 0; i < s.length; i++) {

        hash[s[i]] = i

    }

    let result = []

    let left = 0

    let right = 0

    for (let i = 0; i < s.length; i++) {

        right = Math.max(right, hash[s[i]])

        if (right === i) {

            result.push(right - left + 1)

            left = i + 1

        }

    }

    return result

};

合并区间

var merge = function (intervals) {

  intervals.sort((a, b) => a[0] - b[0]) // 左排序

  let prev = intervals[0]

  let result = []

  for (let i = 0; i < intervals.length; i++) {

    let cur = intervals[i]

    if (cur[0] > prev[1]) {

      // 没有重叠，加入到结果result中

      result.push(prev)

      prev = cur // 更新 prev

    } else {

      // 有重叠 合并

      prev[1] = Math.max(cur[1], prev[1])

    }

  }

  result.push(prev) // 最后一个单独加入

  return result

}

/**

 * @param {number} n

 * @return {number}

 */

var monotoneIncreasingDigits = function (n) {

    let arr = n.toString().split('').map(item => +item)

    let flag = Infinity

    // 从后向前遍历

    for (let i = arr.length - 1; i > 0; i--) {

        if (arr[i - 1] > arr[i]) {

            flag = i // 从哪一位开始标记为9  100 --> 90 

            arr[i - 1] = arr[i - 1] - 1

            arr[i] = 9

        }

    }

    for (let i = flag; i < arr.length; i++) { // 90 --> 99

        arr[i] = 9

    }

    return +arr.join('')

};
